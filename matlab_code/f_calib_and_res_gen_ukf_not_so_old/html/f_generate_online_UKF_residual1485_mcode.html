1function [residual_ukf, residual_ukf_mean, residual_ukf_Variance] = f_generate_online_UKF_residual(Probe_Ampl,Probe_Phase, Forw_Ampl, ...
 2    Forw_Phase, Refl_Ampl, Refl_Probe, FS, F0, cal_coeff, tau_m, K_m, X0, DELAY, FILLING, FLATTOP)
 3% inputs:
 4% Probe_Ampl: Amplitude of Probe measurement
 5% Probe_Phase: Phase of Probe measurement
 6% Forw_Ampl: Amplitude of Forward measurement
 7% Forw_Phase: Phase of Forward measurement
 8% Refl_Ampl: Amplitude of Reflected measurement
 9% Refl_Probe: Phase of Reflected measurement
10% fs: Sampling frequency, for online Data 9MHz, DAQ data 1MHz - can be read
11% from DOOCS property
12% f0: Resonance frequency, should be 1.3GHz - should be read
13% from DOOCS property
14% cal_coeff: coefficients necessary for the calibration of the signals,
15% obtained from the function f_get_calCoeff
16% Rise: Sample numbers that belong to the rise (filling) of the Pulse - should be read
17% from DOOCS property
18% Flattop: Samople numbers that belong to the flattop of the pulse - should be read
19% from DOOCS property
20% Decay: Sample numbers that belong to the decay of the pulse  - should be read
21% from DOOCS property
22%#cogegen
23
24fs = FS*1e6;
25f0 = F0*1e6;
26assert(f0==1.3e9, 'wrong f0 frequency')
27
28if fs ==1e6
29    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
30    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
31    Decay = Flattop(end)+1 : round(2^(14)/9+1);
32else
33    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
34    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
35    Decay = Flattop(end)+1 : 2^(14);
36end
37
38y_mC = complex(Probe_Ampl.*exp(1i*Probe_Phase/180*pi)); % Complex Probe measurement
39u_mC = complex(Forw_Ampl.*exp(1i*Forw_Phase/180*pi)); % Complex Forward measurement
40r_mC = complex(Refl_Ampl.*exp(1i*Refl_Probe/180*pi)); % Reflected
41
42QL  = f_computeQL(Decay, y_mC, fs, f0);
43%% Online residual generation
44initialStateGuess = [0;0;X0];
45MeasNoise_mean = [0;0];
46MeasNoiseVar = diag([0.01, 0.01])*1e-3;
47ProcessVar = diag([0.01, 0.01, 0.01, 0.01, 0.01, 0.01].*1e-3);
48[u_mCal, ~] = f_calibrateData(u_mC, r_mC, y_mC, cal_coeff);
49y_Meas = [real(y_mC).'; imag(y_mC).'];
50u_Meas = [real(u_mCal).'; imag(u_mCal).'];
51residual_ukf_Variance = zeros(2,2,3);
52[res_full_mean, ~, res_full_xi, res_full_Info ] = f_compute_ukf_residual(initialStateGuess, ...
53    MeasNoise_mean, MeasNoiseVar, ProcessVar, y_Meas, u_Meas, f0, fs, QL, tau_m, K_m );
54
55% figure(98)
56% plot(res_full_mean(1,1:9:end),'k.', 'MarkerSize', 8)
57% hold on
58% plot(res_full_mean(2,1:9:end),'g.', 'MarkerSize', 8)
59% xlabel('Samples')
60% ylabel('Residual')
61% legend('I', 'Q')
62% axis tight
63% ylim([-0.025,0.025])
64% ax= gca;
65% ax.FontName = 'Calibri';
66% ax.FontSize = 20;
67% set(gcf,'Color', 'w')
68% Arithmetic mean of the mean
69
70residual_ukf = [mean(res_full_mean(:,Rise),2),...
71    mean(res_full_mean(:,Flattop),2),mean(res_full_mean(:,Decay(1:end-1)),2)];
72
73% Equality Node: Taking the variance into consideration
74residual_ukf_Variance(:,:,1) = [reshape(sum(res_full_Info(Rise,:,:)),[2,2])]\eye(2,2);
75residual_ukf_Variance(:,:,2) = [reshape(sum(res_full_Info(Flattop,:,:)),[2,2])]\eye(2,2);
76
77
78residual_ukf_Variance(:,:,3) = ([reshape(sum(res_full_Info(Decay(1:end-1),:,:)),[2,2])])\eye(2,2);
79
80
81
82residual_ukf_mean = [ [reshape(sum(res_full_Info(Rise,:,:)),[2,2])]\eye(2,2) * sum(res_full_xi(:,Rise),2),...
83    [reshape(sum(res_full_Info(Flattop,:,:)),[2,2])]\eye(2,2)*sum(res_full_xi(:,Flattop),2),...
84    [reshape(sum(res_full_Info(Decay(1:end-1),:,:)),[2,2])]\eye(2,2)*sum(res_full_xi(:,Decay(1:end-1)),2)];
85
86
87end
