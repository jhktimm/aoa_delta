1function [dx]=f_cavitySystem_fixdw(x, u_mC, f0, fs, dw, QL)
 2%#codegen
 3%% Coupled simulink model
 4Ts = 1/fs;
 5w12 = pi*f0./QL;
 6% ------------- Discrete State Space System -----------------------
 7dx = zeros(2,1);
 8% ---------- Values---------
 9a0 = -1+ w12*Ts;
10b0 = 2*w12*Ts;
11u1 = u_mC(1);
12u2 = u_mC(2);
13dx(1) = -a0 * x(1)      - dw*x(2)*Ts + b0 * u1;
14dx(2) =  dw * x(1) * Ts - a0*x(2)    + b0 * u2;
15end
