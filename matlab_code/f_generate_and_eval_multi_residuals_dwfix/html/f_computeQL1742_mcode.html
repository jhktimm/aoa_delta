1function [ QL ] = f_computeQL( Rise, u_mCal, y_mC, fs, f0 )
 2% The quality factor of the cavity is determined by the decay time of the
 3% probes signal. Pulse cuts are considered, as the decay time is deduced
 4% from the pulses.
 5if fs > 1e6
 6    rampup_smpls = 3 * 10;
 7else
 8    rampup_smpls =3;
 9end
10
11idx_u_mCal_cut = zeros(1,size(u_mCal,2));
12for ii = 1:size(u_mCal,2)
13idx_u_mCal_off = find(abs(u_mCal(:,ii))<1.5);
14idx_u_mCal_cutoff = idx_u_mCal_off(idx_u_mCal_off>Rise(1)+rampup_smpls);
15idx_u_mCal_cut(1,ii) = idx_u_mCal_cutoff(1);
16end
17decay_length = length(max(idx_u_mCal_cut):size(u_mCal,1))-1;
18
19lnA = zeros(decay_length+1,size(u_mCal,2));
20for ii = 1:size(u_mCal,2)
21lnA(:,ii) = log(abs(y_mC(idx_u_mCal_cut(1,ii) : idx_u_mCal_cut(1,ii) + decay_length,ii)));
22end
23
24tau = abs(1./((lnA(end,:)-lnA(1,:))/(length(lnA(:,1))./fs)));
25QL = pi*f0*tau;
26

35end
