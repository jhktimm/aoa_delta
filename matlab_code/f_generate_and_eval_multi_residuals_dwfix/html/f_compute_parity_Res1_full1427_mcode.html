1function [ r_1 ] = f_compute_parity_Res1_full(u_mC, y_mC, fs, f0, QL)
 2% f_computeR1 computes the residual- denoted by residual 1- as defined in
 3% the Safeprocess paper.
 4%#codegen
 5%-------------- input and states of the system---------------
 6u1 = real(u_mC);
 7u2 = imag(u_mC);
 8x1 = real(y_mC);
 9x2 = imag(y_mC);
10Ts = 1/fs;
11w12 = pi*f0./QL;
12% ------------------- first residual-------------------------
13x_d1 = -x1(3:end,:) + x1(2:end-1,:) + (-Ts * x1(2:end-1,:) + 2 * Ts * u1(2:end-1,:).* w12);
14x_d2 = (x2(3:end,:) - x2(2:end-1,:) + (Ts * x2(2:end-1,:) - 2* Ts* u2(2:end-1,:)).* w12);
15r_1  = (x_d1.* (Ts.*x1(2:end-1,:)) - x_d2.*(Ts .* x2(2:end-1,:)));
16
17end
