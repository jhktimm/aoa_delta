1function [residual1] = f_generate_online_parityRes1_full(Probe_Ampl,Probe_Phase, Forw_Ampl, ...
 2    Forw_Phase, Refl_Ampl, Refl_Probe, FS, F0, cal_coeff,QL_nom, DELAY, FILLING, FLATTOP)
 3% inputs:
 4% Probe_Ampl: Amplitude of Probe measurement
 5% Probe_Phase: Phase of Probe measurement
 6% Forw_Ampl: Amplitude of Forward measurement
 7% Forw_Phase: Phase of Forward measurement
 8% Refl_Ampl: Amplitude of Reflected measurement
 9% Refl_Probe: Phase of Reflected measurement
10% fs: Sampling frequency, for online Data 9MHz, DAQ data 1MHz - can be read
11% from DOOCS property
12% f0: Resonance frequency, should be 1.3GHz - should be read
13% from DOOCS property
14% cal_coeff: coefficients necessary for the calibration of the signals,
15% obtained from the function f_get_calCoeff
16% Rise: Sample numbers that belong to the rise (filling) of the Pulse - should be read
17% from DOOCS property
18% Flattop: Samople numbers that belong to the flattop of the pulse - should be read
19% from DOOCS property
20% Decay: Sample numbers that belong to the decay of the pulse  - should be read
21% from DOOCS property
22fs = FS*1e6;
23f0 = F0*1e6;
24assert(f0==1.3e9, 'wrong f0 frequency')
25
26if fs ==1e6
27    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
28    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
29    Decay = Flattop(end)+1 : round(2^(14)/9+1);
30else
31    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
32    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
33    Decay = Flattop(end)+1 : 2^(14);
34end
35y_mC = Probe_Ampl.*exp(1i*Probe_Phase/180*pi); % Complex Probe measurement
36u_mC = Forw_Ampl.*exp(1i*Forw_Phase/180*pi); % Complex Forward measurement
37r_mC = Refl_Ampl.*exp(1i*Refl_Probe/180*pi); % Reflected
38
39%% Online residual generation
40
41residual1 = zeros(size(y_mC,1)-2,1);
42% [y_mC, u_mC, r_mC] = f_getData(facility, station, module, cavity);
43[u_mCal, ~] = f_calibrateData(u_mC, r_mC, y_mC, cal_coeff);
44
45residual1(:,1) = f_compute_parity_Res1_full(u_mC, y_mC, fs, f0, QL_nom);
46
47
48end
