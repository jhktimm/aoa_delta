1function [residual2] = f_generate_online_parityRes2_full(Probe_Ampl,Probe_Phase, Forw_Ampl, ...
 2    Forw_Phase, Refl_Ampl, Refl_Probe, FS, F0, cal_coeff, tau_m, K_m,QL_nom,  DELAY, FILLING, FLATTOP)
 3% inputs:
 4% Probe_Ampl: Amplitude of Probe measurement
 5% Probe_Phase: Phase of Probe measurement
 6% Forw_Ampl: Amplitude of Forward measurement
 7% Forw_Phase: Phase of Forward measurement
 8% Refl_Ampl: Amplitude of Reflected measurement
 9% Refl_Probe: Phase of Reflected measurement
10% fs: Sampling frequency, for online Data 9MHz, DAQ data 1MHz - can be read
11% from DOOCS property
12% f0: Resonance frequency, should be 1.3GHz - should be read
13% from DOOCS property
14% cal_coeff: coefficients necessary for the calibration of the signals,
15% obtained from the function f_get_calCoeff
16% Rise: Sample numbers that belong to the rise (filling) of the Pulse - should be read
17% from DOOCS property
18% Flattop: Samople numbers that belong to the flattop of the pulse - should be read
19% from DOOCS property
20% Decay: Sample numbers that belong to the decay of the pulse  - should be read
21% from DOOCS property
22% #cogegen
23fs = FS*1e6;
24f0 = F0*1e6;
25assert(f0==1.3e9, 'wrong f0 frequency')
26
27if fs ==1e6
28    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
29    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
30    Decay = Flattop(end)+1 : round(2^(14)/9+1);
31else
32    Rise = round(DELAY * fs *1e-6):round(DELAY * fs*1e-6+ FILLING*fs*1e-6);
33    Flattop = Rise(end)+1 : Rise(end)+1+round(FLATTOP*fs*1e-6);
34    Decay = Flattop(end)+1 : 2^(14);
35end
36
37y_mC = complex(Probe_Ampl.*exp(1i*Probe_Phase/180*pi)); % Complex Probe measurement
38u_mC = complex(Forw_Ampl.*exp(1i*Forw_Phase/180*pi)); % Complex Forward measurement
39r_mC = complex(Refl_Ampl.*exp(1i*Refl_Probe/180*pi)); % Reflected
40
41%% Online residual generation
42[u_mCal, ~] = f_calibrateData(u_mC, r_mC, y_mC, cal_coeff);
43
44[ residual2 ] = f_compute_parity_Res2_full(u_mCal, y_mC, tau_m, K_m, QL_nom, fs, ...
45                                        f0);
46
47end
