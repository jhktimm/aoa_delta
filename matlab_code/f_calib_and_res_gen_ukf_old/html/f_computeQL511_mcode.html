1function [ QL ] = f_computeQL( Decay, y_mC, fs, f0 )
 2% The quality factor of the cavity is determined by the decay time of the
 3% probes signal.
 4    decay_length = length(Decay);
 5    t2 = int16(Decay(1)+ decay_length/2);
 6    offset_from_end_flattop1 = int16(decay_length*15/32);
 7 
 8    lnA = log(abs(y_mC(t2 : t2 + offset_from_end_flattop1,:)));
 9    tau = abs(1./((lnA(end,:)-lnA(1,:))/(length(lnA(:,1))./fs)));
10    QL = pi*f0*tau;
