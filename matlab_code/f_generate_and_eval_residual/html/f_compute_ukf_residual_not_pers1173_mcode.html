1function [residual_mean, residual_Variance, residual_xi, residual_Info] = f_compute_ukf_residual_not_pers(initialStateGuess, ...
 2    MeasNoise_mean, MeasNoiseVar, ProcessVar, y_Meas, u_Meas, f0, fs, QL, tau_m, K_m )
 3%%
 4%#codegen
 5% Initialize the object.
 6ukf = unscentedKalmanFilter(...
 7    @f_cavitySystem,... % State transition function
 8    @f_CavityMeasurementNonAdditiveNoiseFcn,... % Measurement function
 9    initialStateGuess);
10% Construct the filter
11ukf.MeasurementNoise = MeasNoiseVar;
12ukf.ProcessNoise = ProcessVar;
13ukf.Alpha = 1e-2;
14% end
15[Nstates, ~] = size(ProcessVar); 
16[Noutput, Nsamples] = size(y_Meas);
17xCorrectedUKF = zeros(Nsamples, Nstates);
18PCorrected = zeros(Nsamples, Nstates, Nstates);
19residual_mean = zeros(Noutput, Nsamples);
20residual_Variance = zeros(Nsamples, Noutput, Noutput);
21residual_Info = zeros(Nsamples, Noutput, Noutput);
22residual_xi = zeros(Noutput, Nsamples);
23
24C = zeros(Noutput,Nstates);
25C(1,1) = 1;
26C(2,2) = 1;
27for k=1:Nsamples
28    % Let k denote the current time.
29    %
30    % Residuals (or innovations): Measured output - Predicted output
31    residual_mean(:,k) = y_Meas(:,k) - f_CavityMeasurementNonAdditiveNoiseFcn(ukf.State, MeasNoise_mean); % ukf.State is x[k|k-1] at this point
32    % Incorporate the measurements at time k into the state estimates by
33    % using the "correct" command. This updates the State and StateCovariance
34    % properties of the filter to contain x[k|k] and P[k|k]. These values
35    % are also produced as the output of the "correct" command.
36    [xCorrectedUKF(k,:), PCorrected] = correct(ukf, y_Meas(:,k), MeasNoise_mean);
37    % Predict the states at next time step, k+1. This updates the State and
38    % StateCovariance properties of the filter to contain x[k+1|k] and
39    % P[k+1|k]. These will be utilized by the filter at the next time step.
40    residual_Variance(k, :, :) = C * PCorrected * C.'+ MeasNoiseVar;
41    residual_Info(k, :, :) = (C * PCorrected * C.'+ MeasNoiseVar)\eye(2,2);
42    residual_xi(:,k) = (C * PCorrected * C.'+ MeasNoiseVar)\eye(2,2) * residual_mean(:,k);
43     
44    predict(ukf, u_Meas(:,k), f0, fs, tau_m, K_m, QL);
45end
