1function [ QL ] = f_computeQL( Rise, u_mCal, y_mC, fs, f0 )
 2% The quality factor of the cavity is determined by the decay time of the
 3% probes signal. Pulse cuts are considered, as the decay time is deduced
 4% from the pulses.
 5idx_u_mCal_cut = zeros(1,size(u_mCal,2));
 6for ii = 1:size(u_mCal,2)
 7idx_u_mCal_off = find(abs(u_mCal(:,ii))<1.5);
 8idx_u_mCal_cutoff = idx_u_mCal_off(idx_u_mCal_off>Rise(1)+3);
 9idx_u_mCal_cut(1,ii) = idx_u_mCal_cutoff(1);
10end
11decay_length = length(max(idx_u_mCal_cut):size(u_mCal,1))-1;
12
13lnA = zeros(decay_length+1,size(u_mCal,2));
14for ii = 1:size(u_mCal,2)
15lnA(:,ii) = log(abs(y_mC(idx_u_mCal_cut(1,ii) : idx_u_mCal_cut(1,ii) + decay_length,ii)));
16end
17
18tau = abs(1./((lnA(end,:)-lnA(1,:))/(length(lnA(:,1))./fs)));
19QL = pi*f0*tau;
20
